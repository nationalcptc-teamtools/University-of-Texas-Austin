\newcommand{\setcurcolor}[1]{%
    \ifthenelse{\equal{#1}{4}}{\renewcommand{\curfgcolor}{criticalfg}\renewcommand{\curbgcolor}{criticalbg}\renewcommand{\curboxbgcolor}{criticalboxbg}}{%
        \ifthenelse{\equal{#1}{3}}{\renewcommand{\curfgcolor}{highfg}\renewcommand{\curbgcolor}{highbg}\renewcommand{\curboxbgcolor}{highboxbg}}{%
            \ifthenelse{\equal{#1}{2}}{\renewcommand{\curfgcolor}{mediumfg}\renewcommand{\curbgcolor}{mediumbg}\renewcommand{\curboxbgcolor}{mediumboxbg}}{%
                \ifthenelse{\equal{#1}{1}}{\renewcommand{\curfgcolor}{lowfg}\renewcommand{\curbgcolor}{lowbg}\renewcommand{\curboxbgcolor}{lowboxbg}}{%
                    \ifthenelse{\equal{#1}{0}}{\renewcommand{\curfgcolor}{infofg}\renewcommand{\curbgcolor}{infobg}\renewcommand{\curboxbgcolor}{infoboxbg}}{%
                        \renewcommand{\curcolor}{black}}
                }}}}
}
